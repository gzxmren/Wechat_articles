# skills/excel-analyzer/utils/parser.py

class ExcelTableParser:
    def __init__(self, file_path):
        self.file_path = file_path
        self.wb = load_workbook(file_path, data_only=True)
    
    # 🔑 核心逻辑（只需看这里）
    def extract_all_sheets(self):
        """提取所有工作表数据"""
        sheets_data = {}
        
        for sheet_name in self.wb.sheetnames:
            sheet = self.wb[sheet_name]
            data = {
                'name': sheet_name,
                'data': [],
                'formulas': [],
                'metadata': {
                    'rows': sheet.max_row,
                    'columns': sheet.max_column,
                }
            }
            
            # 提取数据
            for row in sheet.iter_rows(values_only=True):
                data['data'].append(row)
            
            # 提取公式
            for row in sheet.iter_rows():
                for cell in row:
                    if cell.value and str(cell.value).startswith('='):
                        data['formulas'].append({
                            'cell': cell.coordinate,
                            'formula': cell.value,
                        })
            
            sheets_data[sheet_name] = data
        
        return sheets_data

def check_formulas(self):
    """检查公式错误"""
    errors = []
    for sheet in self.wb.sheetnames:
        ws = self.wb[sheet]
        for row in ws.iter_rows():
            for cell in row:
                # ← 关键判断
                if cell.value.startswith('='):  
                    if self._has_error(cell.value):
                        errors.append({
                            'cell': cell.coordinate,
                            'formula': cell.value,
                            'error': self._get_error_type(cell.value)
                        })
    return {'total': len(errors), 'errors': errors}

支持的错误类型：

  • 语法错误：括号不匹配、函数拼写错误
  • 引用错误：引用范围无效、循环引用
  • 逻辑错误：除零、类型不匹配
  • 优化建议：可简化为更高效公式

# skills/excel-analyzer/utils/anomalies.py

# 🔑 核心逻辑（只需看这里）
def detect_anomalies(self, column=None):
    """检测数据异常 - Z-Score 方法"""
    anomalies = []
    numeric_cols = self.df.select_dtypes(include=['number']).columns
    
    for col in (column and [column] or numeric_cols):
        values = self.df[col].dropna()
        if len(values) < 3:
            continue
        
        # Z-Score 检测
        mean, std = values.mean(), values.std()
        for idx, val in enumerate(values):
            z_score = abs((val - mean) / std)
            # ← 关键判断：超过 3 个标准差
            if z_score > 3:  
                anomalies.append({
                    'column': col,
                    'row': idx + 2,
                    'value': val,
                    'reason': f'Z-Score: {z_score:.2f}'
                })
    
    return {'total': len(anomalies), 'anomalies': anomalies}

async def analyze_root_cause(self, anomaly, historical_data):
    """分析异常背后的原因"""
    
    # 1. 多维度对比分析
    comparisons = {
        'yoy': self.compare_with_same_period_last_year(anomaly, historical_data),
        'mom': self.compare_with_last_month(anomaly, historical_data),
        'vs_target': self.compare_with_target(anomaly),
        'vs_peer': self.compare_with_peers(anomaly, historical_data)
    }

    factors = await sessions_spawn({
      task: f"""分析以下数据异常的可能原因：
      
异常数据：{json.dumps(anomaly)}
对比分析：{json.dumps(comparisons)}

请从以下维度分析：
1. 内部因素（数据录入、计算逻辑、业务调整）
2. 外部因素（季节性、市场环境、政策变化）

    return {

        'comparisons': comparisons,

### 2. main.py（主入口）

```python
#!/usr/bin/env python3
"""
Excel 智能分析技能 - Tool + Advisor 模式

功能：
1. Tool: 公式检查、异常检测、数据清洗
2. Advisor: 图表建议、优化建议、业务洞察
3. Advanced: 自然语言查询、根因分析、智能预警、数据故事、What-If 推演
"""

from openpyxl import load_workbook
import pandas as pd
import json
import sys
import asyncio

# 导入工具模块
from utils.parser import ExcelTableParser
from utils.formulas import FormulaChecker
from utils.anomalies import AnomalyDetector
from utils.suggestions import ChartSuggester
from utils.nl-query import NaturalLanguageQuery
from utils.root-cause import RootCauseAnalyzer
from utils.alert import AlertManager
from utils.story import StoryGenerator
from utils.whatif import WhatIfAnalyzer


class ExcelAnalyzer:
    """Excel 智能分析主类 - 统一入口"""
    
    def __init__(self, file_path):
        self.file_path = file_path
        self.wb = load_workbook(file_path)
        self.df = pd.read_excel(file_path)
        
        # 初始化工具模块
        self.parser = ExcelTableParser(file_path)
        self.formula_checker = FormulaChecker(self.wb)
        self.anomaly_detector = AnomalyDetector(self.df)
        self.chart_suggester = ChartSuggester(self.df)
        self.nl_query = NaturalLanguageQuery(self.df)
        self.root_cause_analyzer = RootCauseAnalyzer(self.df)
        self.alert_manager = AlertManager(file_path)
        self.story_generator = StoryGenerator()
        self.whatif_analyzer = WhatIfAnalyzer(self.df)
    
    # ========== Tool 模式：执行具体任务 ==========
    
    def check_formulas(self):
        """功能 2：检查公式错误"""
        return self.formula_checker.check()
    
    def detect_anomalies(self, column=None):
        """功能 3：检测数据异常"""
        return self.anomaly_detector.detect(column)
    
    def parse_excel(self):
        """功能 1：解析 Excel 文件"""
        return self.parser.extract_all_sheets()
    
    # ========== Advisor 模式：提供建议 ==========
    
    def suggest_charts(self):
        """图表建议"""
        return self.chart_suggester.suggest()
    
    def business_insights(self):
        """业务洞察"""
        return self.chart_suggester.get_insights()
    
    # ========== 高级功能 ==========
    
    async def query(self, question):
        """功能 4：自然语言查询"""
        return await self.nl_query.query(question)
    
    async def analyze_root_cause(self, anomaly):
        """功能 5：根因分析"""
        return await self.root_cause_analyzer.analyze(anomaly)
    
    async def subscribe_alert(self, rules):
        """功能 6：设置预警"""
        return await self.alert_manager.subscribe(rules)
    
    async def generate_story(self, audience='管理层', style='专业'):
        """功能 7：数据故事生成"""
        insights = self.business_insights()
        return await self.story_generator.generate(insights, audience, style)
    
    async def what_if(self, questions):
        """功能 8：What-If 沙盘推演"""
        base_scenario = self._get_base_scenario()
        return await self.whatif_analyzer.analyze(base_scenario, questions)
    
    def _get_base_scenario(self):
        """获取基础场景数据"""
        return {
            'revenue': self.df.select_dtypes(include=['number']).iloc[-1].sum(),
            'growth_rate': self._calculate_growth_rate(),
            'metrics': self.df.describe()
        }
    
    def _calculate_growth_rate(self):
        """计算增长率"""
        numeric_cols = self.df.select_dtypes(include=['number'])
        if len(numeric_cols) > 0 and len(numeric_cols) >= 2:
            first = numeric_cols.iloc[0].sum()
            last = numeric_cols.iloc[-1].sum()
            return (last - first) / first if first != 0 else 0
        return 0
    
    # ========== 统一入口 ==========
    
    def analyze(self, options=None):
        """完整分析（默认入口）"""
        options = options or {}
        return {
            'file': self.file_path,
            'sheets': len(self.wb.sheetnames),
            'rows': len(self.df),
            'columns': len(self.df.columns),
            'formula_check': self.check_formulas(),
            'anomalies': self.detect_anomalies(),
            'chart_suggestions': self.suggest_charts(),
            'insights': self.business_insights()
        }
    
    async def analyze_advanced(self, options=None):
        """高级分析（包含 AI 功能）"""
        options = options or {}
        base = self.analyze(options)
        
        # 添加高级功能
        if options.get('nl_query'):
            base['nl_query'] = await self.query(options['nl_query'])
        if options.get('root_cause'):
            base['root_cause'] = await self.analyze_root_cause(options['root_cause'])
        if options.get('alerts'):
            base['alerts'] = await self.subscribe_alert(options['alerts'])
        if options.get('story'):
            base['story'] = await self.generate_story()
        if options.get('whatif'):
            base['whatif'] = await self.what_if(options['whatif'])
        
        return base

⚠️ 常见陷阱与解决方案

坑 1：大文件处理慢

问题：10 万 + 行数据，分析要 5 分钟

解决：

# config.yaml
performance:
  chunk_size: 5000  # 分批处理
  parallel: true    # 并行处理

效果：从 5 分钟降至 30 秒

坑 2：中文列名识别错误

问题：列名是中文，异常检测失效

解决：

# 添加列名映射
column_mapping = {
    '销售额': 'sales',
    '利润': 'profit',
    '增长率': 'growth_rate'
}

坑 3：预测结果不准确

问题：预测偏差太大

解决：

  1. 增加历史数据量（至少需要 12 期数据）
  2. 检查数据是否有异常值
  3. 使用集成预测（默认）
./scripts/analyze-excel.sh file.xlsx --ensemble
